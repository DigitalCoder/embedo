embedo

A simple, fast, lightweight and automagical social media content embedder plugin for web

Star Download

Implementation JS code example:

        
      /**
       * Eithe import the script source
       * OR npm install embedo
       * OR bower install embedo
       */
      var embedo = new Embedo({
        facebook: true,
        twitter: true,
        instagram: true
      });

      // Basic embedo load call
      embedo.load(
        document.getElementById('my-awesome-container'),
        'https://twitter.com/Sh0bhit/status/830448643379458048'
      );

      // Passing custom width and height
      embedo.load(
        document.getElementById('my-awesome-container'),
        'https://twitter.com/Sh0bhit/status/830448643379458048',
        {
          width: 640,
          height: 480
        }
      );

      // Load without automagic
      embedo.load(
        document.getElementById('my-awesome-container'),
        'https://twitter.com/Sh0bhit/status/830448643379458048',
        {
          strict: true
        }
      );
